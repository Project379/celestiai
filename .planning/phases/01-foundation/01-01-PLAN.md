<task type="auto">
  <name>Task 1: Initialize Turborepo monorepo structure</name>
  <files>
    turbo.json
    pnpm-workspace.yaml
    package.json
    .gitignore
    tsconfig.json
  </files>
  <action>
Create monorepo root files manually (do NOT use create-turbo as it scaffolds apps we don't need):

1. Create `package.json`:
```json
{
  "name": "celestia-ai",
  "private": true,
  "scripts": {
    "dev": "turbo run dev",
    "build": "turbo run build",
    "lint": "turbo run lint",
    "typecheck": "turbo run typecheck"
  },
  "devDependencies": {
    "turbo": "^2.4.4",
    "typescript": "^5.7.3"
  },
  "packageManager": "pnpm@9.15.4"
}
```

2. Create `turbo.json`:
```json
{
  "$schema": "https://turborepo.dev/schema.json",
  "tasks": {
    "build": {
      "dependsOn": ["^build"],
      "outputs": [".next/**", "!.next/cache/**", "dist/**"]
    },
    "dev": {
      "cache": false,
      "persistent": true
    },
    "lint": {},
    "typecheck": {
      "dependsOn": ["^typecheck"]
    }
  }
}
```

3. Create `pnpm-workspace.yaml`:
```yaml
packages:
  - 'apps/*'
  - 'packages/*'
```

4. Create root `tsconfig.json`:
```json
{
  "compilerOptions": {
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "moduleResolution": "bundler",
    "resolveJsonModule": true,
    "isolatedModules": true,
    "noEmit": true
  }
}
```

5. Update `.gitignore` to include:
```
node_modules
.turbo
.next
dist
.env*.local
```
  </action>
  <verify>
Files exist: `turbo.json`, `pnpm-workspace.yaml`, `package.json`, `tsconfig.json`
  </verify>
  <done>
Root monorepo configuration files created with Turborepo 2.x task syntax.
  </done>
</task>

  <action>
Create the Next.js 15 web app in `apps/web/`:

1. Create `apps/web/package.json`:
```json
{
  "name": "@celestia/web",
  "version": "0.1.0",
  "private": true,
  "scripts": {
    "dev": "next dev --turbopack",
    "build": "next build",
    "start": "next start",
    "lint": "next lint",
    "typecheck": "tsc --noEmit"
  },
  "dependencies": {
    "next": "^15.2.4",
    "react": "^19.0.0",
    "react-dom": "^19.0.0"
  },
  "devDependencies": {
    "@types/node": "^22.10.10",
    "@types/react": "^19.0.8",
    "@types/react-dom": "^19.0.3",
    "typescript": "^5.7.3"
  }
}
```

Note: Using Next.js 15.2.4+ (patched for CVE-2025-29927 middleware vulnerability).

2. Create `apps/web/next.config.js`:
```javascript
/** @type {import('next').NextConfig} */
const nextConfig = {
  reactStrictMode: true,
}

module.exports = nextConfig
```

3. Create `apps/web/tsconfig.json`:
```json
{
  "extends": "../../tsconfig.json",
  "compilerOptions": {
    "lib": ["dom", "dom.iterable", "esnext"],
    "module": "esnext",
    "jsx": "preserve",
    "incremental": true,
    "plugins": [{ "name": "next" }],
    "paths": {
      "@/*": ["./*"]
    }
  },
  "include": ["next-env.d.ts", "**/*.ts", "**/*.tsx", ".next/types/**/*.ts"],
  "exclude": ["node_modules"]
}
```

4. Create `apps/web/app/layout.tsx`:
```tsx
import type { Metadata } from 'next'

export const metadata: Metadata = {
  title: 'Celestia AI',
  description: 'Вашият астрологичен спътник',
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="bg">
      <body>{children}</body>
    </html>
  )
}
```

5. Create `apps/web/app/page.tsx`:
```tsx
export default function HomePage() {
  return (
    <main>
      <h1>Celestia AI</h1>
      <p>Добре дошли</p>
    </main>
  )
}
```

Create directories as needed: `apps/web/app/`
  </action>
